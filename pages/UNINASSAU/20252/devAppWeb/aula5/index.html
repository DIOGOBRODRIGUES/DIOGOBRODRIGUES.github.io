    // Configura√ß√µes padr√£o

// Plugin mais avan√ßado - Contador animado
(function($) {
  $.fn.contadorAnimado = function(options) {
    var settings = $.extend({
      inicio: 0,
      fim: 100,
      duracao: 2000,
      sufixo: ''
    }, options);
    
    return this.each(function() {
      var $this = $(this);
      var inicio = settings.inicio;
      var fim = settings.fim;
      
      $({ counter: inicio }).animate({ counter: fim }, {
        duration: settings.duracao,
        step: function() {
          $this.text(Math.ceil(this.counter) + settings.sufixo);
        },
        complete: function() {
          $this.text(fim + settings.sufixo);
        }
      });
    });
  };
})(jQuery);
