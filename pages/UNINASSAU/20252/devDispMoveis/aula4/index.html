🤔 O que é React Native?

  • 📱 Framework para criar apps móveis nativos
  • ⚛️ Usa JavaScript e React que já conhecemos
  • 🏢 Desenvolvido pelo Facebook (Meta)
  • 🔄 Um código para Android e iOS
  • 🎯 Apps realmente nativos (não webview)

Diferencial: Componentes viram elementos nativos!

⚙️ React Native vs React Web

React Web:

  • JSX → HTML DOM
  • CSS para estilos
  • Navegador renderiza
  • div, span, p, img

React Native:

  • JSX → Componentes Nativos
  • StyleSheet JS
  • iOS/Android renderiza
  • View, Text, Image

🚀 Por que usar Expo?

  • ⚡ Setup rápido: Sem configuração complexa
  • 📱 Teste imediato: No celular via QR Code
  • 🌐 Desenvolvimento online: Expo Snack
  • 🔄 Hot reload: Mudanças em tempo real
  • 🎯 Ideal para aprender: Foco no código

🌐 Expo Snack - Desenvolva Online!

📝 Input de Texto

import React, { useState } from 'react';
import { View, Text, TextInput, TouchableOpacity, StyleSheet } from 'react-native';

export default function App() {
  const [nome, setNome] = useState('');
  const [mensagem, setMensagem] = useState('');

  const cumprimentar = () => {
    if (nome.trim()) {
      setMensagem(`Olá, ${nome}! Bem-vindo(a)! 👋`);
    } else {
      setMensagem('Por favor, digite seu nome!');
    }
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Qual seu nome?</Text>
      
      <TextInput
        style={styles.input}
        placeholder="Digite seu nome..."
        value={nome}
        onChangeText={setNome}
        placeholderTextColor="#999"
      />
      
      <TouchableOpacity style={styles.botao} onPress={cumprimentar}>
        <Text style={styles.textoBotao}>Cumprimentar</Text>
      </TouchableOpacity>
      
      {mensagem ? (
        <Text style={styles.mensagem}>{mensagem}</Text>
      ) : null}
    </View>
  );
}

import React, { useState } from 'react';
import { View, Text, TouchableOpacity, StyleSheet } from 'react-native';

export default function CartaoVisita() {
  const [cor, setCor] = useState('#007AFF');
  const [visualizacoes, setVisualizacoes] = useState(0);

  const mudarCor = () => {
    const cores = ['#007AFF', '#FF3B30', '#34C759', '#FF9500', '#AF52DE'];
    const novaCor = cores[Math.floor(Math.random() * cores.length)];
    setCor(novaCor);
    setVisualizacoes(visualizacoes + 1);
  };

  return (
    <View style={styles.container}>
      <View style={[styles.cartao, { borderLeftColor: cor }]}>
        <Text style={styles.nome}>João Silva</Text>
        <Text style={styles.profissao}>Desenvolvedor Mobile</Text>
        <Text style={styles.contato}>📧 joao@email.com</Text>
        <Text style={styles.contato}>📱 (11) 99999-9999</Text>
        
        <TouchableOpacity style={[styles.botao, { backgroundColor: cor }]} onPress={mudarCor}>
          <Text style={styles.textoBotao}>Mudar Cor</Text>
        </TouchableOpacity>
        
        <Text style={styles.visualizacoes}>Visualizações: {visualizacoes}</Text>
      </View>
    </View>
  );
}

  const addTask = () => {
    if (task.trim()) {
      setTasks([...tasks, { id: Date.now(), text: task }]);
      setTask('');
    }
  };

  const removeTask = (id) => {
    setTasks(tasks.filter(t => t.id !== id));
  };

  return (
    <SafeAreaView style={todoStyles.container}>
      <Text style={todoStyles.title}>📝 Minhas Tarefas</Text>
      
      <View style={todoStyles.inputContainer}>
        <TextInput
          style={todoStyles.input}
          value={task}
          onChangeText={setTask}
          placeholder="Digite uma tarefa..."
        />
        <TouchableOpacity 
          style={todoStyles.addButton} 
          onPress={addTask}
        >
          <Text style={todoStyles.addButtonText}>+</Text>
        </TouchableOpacity>
      </View>

      <FlatList
        data={tasks}
        renderItem={({ item }) => (
          <View style={todoStyles.taskItem}>
            <Text style={todoStyles.taskText}>{item.text}</Text>
            <TouchableOpacity onPress={() => removeTask(item.id)}>
              <Text>❌</Text>
            </TouchableOpacity>
          </View>
        )}
        keyExtractor={item => item.id.toString()}
      />
    </SafeAreaView>
  );
}
