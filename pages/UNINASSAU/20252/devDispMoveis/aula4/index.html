    import React, { useState } from 'react';
import {
  View, Text, TextInput, TouchableOpacity,
  FlatList, StyleSheet, SafeAreaView,
} from 'react-native';

export default function TodoApp() {
  const [task, setTask] = useState('');
  const [tasks, setTasks] = useState([]);

  const addTask = () => {
    if (task.trim()) {
      setTasks([...tasks, { id: Date.now(), text: task }]);
      setTask('');
    }
  };

  const removeTask = (id) => {
    setTasks(tasks.filter(t => t.id !== id));
  };

  return (
    <SafeAreaView style={todoStyles.container}>
      <Text style={todoStyles.title}>ğŸ“ Minhas Tarefas</Text>
      
      <View style={todoStyles.inputContainer}>
        <TextInput
          style={todoStyles.input}
          value={task}
          onChangeText={setTask}
          placeholder="Digite uma tarefa..."
        />
        <TouchableOpacity 
          style={todoStyles.addButton} 
          onPress={addTask}
        >
          <Text style={todoStyles.addButtonText}>+</Text>
        </TouchableOpacity>
      </View>

      <FlatList
        data={tasks}
        renderItem={({ item }) => (
          <View style={todoStyles.taskItem}>
            <Text style={todoStyles.taskText}>{item.text}</Text>
            <TouchableOpacity onPress={() => removeTask(item.id)}>
              <Text>âŒ</Text>
            </TouchableOpacity>
          </View>
        )}
        keyExtractor={item => item.id.toString()}
      />
    </SafeAreaView>
  );
}
