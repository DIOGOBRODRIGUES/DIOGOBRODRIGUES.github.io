import React, { useState } from 'react';
import { View, Text, StyleSheet, TextInput, FlatList } from 'react-native';
import { Ionicons } from '@expo/vector-icons';

const allDishes = [
  { id: '1', name: 'Pizza Margherita', restaurant: 'Pizza Mania' },
  { id: '2', name: 'Burger ClÃ¡ssico', restaurant: 'Burger House' },
  { id: '3', name: 'Sushi Combo', restaurant: 'Sushi Master' },
  { id: '4', name: 'Lasanha', restaurant: 'Pasta Italia' },
];

export default function SearchScreen() {
  const [searchQuery, setSearchQuery] = useState('');
  const [results, setResults] = useState([]);

  const handleSearch = (text) => {
    setSearchQuery(text);
    if (text.length > 0) {
      const filtered = allDishes.filter(dish =>
        dish.name.toLowerCase().includes(text.toLowerCase())
      );
      setResults(filtered);
    } else {
      setResults([]);
    }
  };

  return (
    <View style={styles.container}>
      <View style={styles.searchContainer}>
        <Ionicons name="search" size={20} color="#666" style={styles.searchIcon} />
        <TextInput
          style={styles.searchInput}
          placeholder="Buscar pratos..."
          value={searchQuery}
          onChangeText={handleSearch}
        />
      </View>

      {results.length > 0 ? (
        <FlatList
          data={results}
          keyExtractor={item => item.id}
          renderItem={({ item }) => (
            <View style={styles.resultItem}>
              <Text style={styles.dishName}>{item.name}</Text>
              <Text style={styles.restaurantName}>{item.restaurant}</Text>
            </View>
          )}
        />
      ) : (
        <View style={styles.emptyContainer}>
          <Ionicons name="restaurant-outline" size={64} color="#ccc" />
          <Text style={styles.emptyText}>
            {searchQuery ? 'Nenhum prato encontrado' : 'Digite para buscar'}
          </Text>
        </View>
      )}
    </View>
  );
}

              <Text style={styles.restaurantName}>{item.restaurant}</Text>
