🔍 JSX vs HTML: As Diferenças

HTML JSX Por quê?
class="botao" className="botao" class é palavra reservada do JS
onclick="funcao()" onClick={funcao} camelCase + função como valor
<input> <input /> Tags devem ser fechadas
for="nome" htmlFor="nome" for é palavra reservada do JS

🎯 JavaScript Dentro do JSX

function ComponenteCompleto() {
  const usuario = {
    nome: "Maria",
    idade: 28,
    profissao: "Desenvolvedora"
  };
  
  const estiloTitulo = {
    color: "#61dafb",
    fontSize: "24px",
    fontWeight: "bold"
  };

  return (
    <div>
      <h1 style={estiloTitulo}>
        Olá, {usuario.nome}! 👋
      </h1>
      <p>Idade: {usuario.idade} anos</p>
      <p>Profissão: {usuario.profissao}</p>
      <p>Ano de nascimento: {2025 - usuario.idade}</p>
      {usuario.idade >= 18 ? (
        <button>✅ Maior de idade</button>
      ) : (
        <button>⚠️ Menor de idade</button>
      )}
    </div>
  );
}

🚨 Regras de Ouro do JSX

✅ Sempre Faça

  • 🔒 Um elemento pai: <div> ou <></>
  • 🏷️ className em vez de class
  • 📝 camelCase nos atributos
  • 🔐 Fechar tags: <img />, <br />
  • 🧮 {} para JavaScript

❌ Nunca Faça

  • 🚫 Múltiplos elementos sem wrapper
  • 🚫 Misturar HTML e JSX
  • 🚫 Usar palavras reservadas do JS
  • 🚫 Esquecer de fechar tags
  • 🚫 Usar string para funções

🧱 Anatomia de um Componente

// 1. 📥 Importações (se necessário)
import React from 'react';
import './BotaoEstiloso.css';

// 2. 🏗️ Função do Componente
function BotaoEstiloso() {
  // 3. 🧠 Lógica (variáveis, funções)
  const corPrimaria = "#61dafb";
  
  const cliquei = () => {
    alert("Botão clicado! 🎉");
  };

  // 4. 📤 Retorno JSX
  return (
    <button 
      className="botao-estiloso"
      style={{ backgroundColor: corPrimaria }}
      onClick={cliquei}
    >
      Clique em mim! 🚀
    </button>
  );
}

// 5. 📤 Exportação
export default BotaoEstiloso;

As 4 Maneiras de Estilizar

1.CSS Tradicional

/* styles.css */
.botao-azul {
  background: #2196f3;
  padding: 10px 20px;
  border: none;
  border-radius: 5px;
}

2.CSS Modules

/* Botao.module.css */
.azul {
  background: #2196f3;
}

// Botao.js
import styles from './Botao.module.css';

3.Inline Styles

const estilos = {
  backgroundColor: '#2196f3',
  padding: '10px 20px',
  border: 'none'
};

<button style={estilos}>

4.Styled Components

import styled from 'styled-components';

const BotaoAzul = styled.button`
  background: #2196f3;
  padding: 10px 20px;
`;

🎯 Quando Usar Cada Um?

Método Quando Usar Prós Contras
CSS Tradicional Projetos simples, estilos globais Familiar, simples Conflitos de nomes
CSS Modules Componentes específicos Escopo local Setup extra
Inline Styles Estilos dinâmicos Dinâmico, rápido Performance, limitado
Styled Components Design systems Poderoso, moderno Curva de aprendizado

🎨 Exemplo Prático: Card de Produto

// CardProduto.js
import './CardProduto.css';

function CardProduto() {
  const produto = {
    nome: "iPhone 15",
    preco: 4999.99,
    imagem: "iphone.jpg",
    disponivel: true
  };

  const estiloPreco = {
    color: produto.disponivel ? '#4caf50' : '#f44336',
    fontWeight: 'bold',
    fontSize: '18px'
  };

  return (
    <div className="card-produto">
      <img src={produto.imagem} alt={produto.nome} />
      <h3>{produto.nome}</h3>
      <p style={estiloPreco}>
        R$ {produto.preco.toFixed(2)}
      </p>
      <button className={produto.disponivel ? 'btn-comprar' : 'btn-indisponivel'}>
        {produto.disponivel ? '🛒 Comprar' : '❌ Indisponível'}
      </button>
    </div>
  );
}

🔄 Import/Export: Organizando o Código

📦 Export: Enviando Componentes

🎯 Export Default (Um por arquivo)

// Botao.js
function Botao() {
  return <button>Click</button>;
}

export default Botao;

// OU direto na função
export default function Botao() {
  return <button>Click</button>;
}

📎 Named Export (Vários por arquivo)

// Botoes.js
export function BotaoPrimario() {
  return <button className="primary">Primary</button>;
}

export function BotaoSecundario() {
  return <button className="secondary">Secondary</button>;
}

export const BotaoPequeno = () => 
  <button className="small">Small</button>;

📥 Import: Recebendo Componentes

// App.js

// 🎯 Import Default
import Botao from './components/Botao';
import CardProduto from './components/CardProduto';

// 📎 Named Imports
import { BotaoPrimario, BotaoSecundario } from './components/Botoes';

// 🔄 Import tudo como namespace
import * as Botoes from './components/Botoes';

// 🎨 Import CSS
import './App.css';

function App() {
  return (
    <div className="App">
      <Botao />
      <BotaoPrimario />
      <BotaoSecundario />
      <Botoes.BotaoPequeno />
      <CardProduto />
    </div>
  );
}

🎯 Vamos Criar: Header de Site

// components/Header/Header.js
import './Header.css';

function Header() {
  const menuItems = ['Home', 'Sobre', 'Serviços', 'Contato'];
  
  return (
    <header className="header">
      <div className="container">
        <div className="logo">
          <h1>🚀 MinhaEmpresa</h1>
        </div>
        
        <nav className="navigation">
          <ul className="menu">
            {menuItems.map((item, index) => (
              <li key={index} className="menu-item">
                <a href={`#${item.toLowerCase()}`}>{item}</a>
              </li>
            ))}
          </ul>
        </nav>
        
        <div className="actions">
          <button className="btn-login">Login</button>
          <button className="btn-signup">Cadastrar</button>
        </div>
      </div>
    </header>
  );
}

export default Header;

🚨 Debugging: Erros Comuns

🐛 Top 5 Erros de Iniciantes

❌ Erro: Múltiplos elementos

// ❌ Não funciona
return (
  <h1>Título</h1>
  <p>Parágrafo</p>
);
// ✅ Funciona
return (
  <>
    <h1>Título</h1>
    <p>Parágrafo</p>
  </>
);

❌ Erro: class em vez de className

// ❌ Não funciona
<div class="container">
  Conteúdo
</div>
// ✅ Funciona
<div className="container">
  Conteúdo
</div>

🔧 Outros Erros Frequentes

3. Tags não fechadas

❌ <img src="foto.jpg">

✅ <img src="foto.jpg" />

4. Função como string

❌ onClick="minhaFuncao()"

✅ onClick={minhaFuncao}

5. Esquecer export

❌ function Botao() {}

✅ export default Botao;
