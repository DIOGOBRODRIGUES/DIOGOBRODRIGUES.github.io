const { createApp } = Vue

createApp({
  data() {
    return {
      novaTarefa: '',
      filtro: 'todos',
      tarefas: [
        { id: 1, texto: 'Aprender Vue.js', concluida: false },
        { id: 2, texto: 'Fazer exerc√≠cios', concluida: true },
        { id: 3, texto: 'Criar projeto', concluida: false }
      ]
    }
  },
  
  computed: {
    tarefasFiltradas() {
      if (this.filtro === 'ativas') {
        return this.tarefas.filter(t => !t.concluida)
      } else if (this.filtro === 'concluidas') {
        return this.tarefas.filter(t => t.concluida)
      }
      return this.tarefas
    },
    
    tarefasAtivas() {
      return this.tarefas.filter(t => !t.concluida)
    }
  },
  
  methods: {
    adicionarTarefa() {
      if (this.novaTarefa.trim() !== '') {
        this.tarefas.push({
          id: Date.now(),
          texto: this.novaTarefa.trim(),
          concluida: false
        })
        this.novaTarefa = ''
      }
    },
    
    removerTarefa(id) {
      this.tarefas = this.tarefas.filter(t => t.id !== id)
    }
  }
}).mount('#app')
