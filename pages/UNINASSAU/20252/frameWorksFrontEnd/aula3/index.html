💻 Atividade - Lógica Vue.js

Script Vue.js para Todo List

const { createApp } = Vue

createApp({
  data() {
    return {
      novaTarefa: '',
      filtro: 'todos',
      tarefas: [
        { id: 1, texto: 'Aprender Vue.js', concluida: false },
        { id: 2, texto: 'Fazer exercícios', concluida: true },
        { id: 3, texto: 'Criar projeto', concluida: false }
      ]
    }
  },
  
  computed: {
    tarefasFiltradas() {
      if (this.filtro === 'ativas') {
        return this.tarefas.filter(t => !t.concluida)
      } else if (this.filtro === 'concluidas') {
        return this.tarefas.filter(t => t.concluida)
      }
      return this.tarefas
    },
    
    tarefasAtivas() {
      return this.tarefas.filter(t => !t.concluida)
    }
  },
  
  methods: {
    adicionarTarefa() {
      if (this.novaTarefa.trim() !== '') {
        this.tarefas.push({
          id: Date.now(),
          texto: this.novaTarefa.trim(),
          concluida: false
        })
        this.novaTarefa = ''
      }
    },
    
    removerTarefa(id) {
      this.tarefas = this.tarefas.filter(t => t.id !== id)
    }
  }
}).mount('#app')

⚙️ Computed Properties

Propriedades calculadas que são reativas e cacheadas

Reativas

Atualizadas quando dependências mudam

Cacheadas

Resultado salvo até mudanças

Performance

Melhor que methods para cálculos

Computed vs Methods

createApp({
  data() {
    return {
      produtos: [
        { nome: 'Notebook', preco: 2000 },
        { nome: 'Mouse', preco: 50 },
        { nome: 'Teclado', preco: 150 }
      ]
    }
  },
  
  computed: {
    // ✅ Computed - Cacheado e reativo
    totalProdutos() {
      console.log('Calculando total...') // Executado apenas quando produtos muda
      return this.produtos.reduce((total, p) => total + p.preco, 0)
    },
    
    produtosCaros() {
      return this.produtos.filter(p => p.preco > 100)
    }
  },
  
  methods: {
    // ❌ Method - Executado a cada chamada
    getTotalProdutos() {
      console.log('Calculando total via method...') // Executado toda vez
      return this.produtos.reduce((total, p) => total + p.preco, 0)
    }
  }
}).mount('#app')

🆕 Composition API - Vue 3

Nova forma de organizar lógica de componentes

Options API vs Composition API

// Options API (Vue 2 style)
export default {
  data() {
    return {
      contador: 0
    }
  },
  methods: {
    incrementar() {
      this.contador++
    }
  },
  mounted() {
    console.log('Montado!')
  }
}

// Composition API (Vue 3)
import { ref, onMounted } from 'vue'

export default {
  setup() {
    const contador = ref(0)
    
    const incrementar = () => {
      contador.value++
    }
    
    onMounted(() => {
      console.log('Montado!')
    })
    
    return {
      contador,
      incrementar
    }
  }
}

💡 Composition API permite melhor reutilização de lógica entre componentes
