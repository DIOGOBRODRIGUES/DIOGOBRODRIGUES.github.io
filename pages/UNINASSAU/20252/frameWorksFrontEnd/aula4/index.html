function Calculadora() {
  const [display, setDisplay] = useState('0');
  const [operacao, setOperacao] = useState(null);
  const [valor, setValor] = useState(null);

  const inserirNumero = (numero) => {
    if (display === '0') {
      setDisplay(numero.toString());
    } else {
      setDisplay(display + numero.toString());
    }
  };

  const inserirOperacao = (op) => {
    setValor(parseFloat(display));
    setOperacao(op);
    setDisplay('0');
  };

  const calcular = () => {
    const atual = parseFloat(display);
    let resultado;
    
    switch (operacao) {
      case '+': resultado = valor + atual; break;
      case '-': resultado = valor - atual; break;
      case '*': resultado = valor * atual; break;
      case '/': resultado = valor / atual; break;
      default: return;
    }
    
    setDisplay(resultado.toString());
    setOperacao(null);
    setValor(null);
  };

  const limpar = () => {
    setDisplay('0');
    setOperacao(null);
    setValor(null);
  };

  return (
    <div style={{ textAlign: 'center', maxWidth: '300px', margin: '0 auto' }}>
      <div style={{ 
        backgroundColor: '#000', 
        color: '#0f0', 
        padding: '20px', 
        fontSize: '24px',
        fontFamily: 'monospace'
      }}>
        {display}
      </div>
      
      <div style={{ display: 'grid', gridTemplateColumns: 'repeat(4, 1fr)', gap: '5px' }}>
        <button onClick={limpar}>C</button>
        <button onClick={() => inserirOperacao('/')}>&divide;</button>
        <button onClick={() => inserirOperacao('*')}&times;</button>
        <button onClick={() => inserirOperacao('-')}&minus;</button>
        
        <button onClick={() => inserirNumero(7)}>7</button>
        <button onClick={() => inserirNumero(8)}>8</button>
        <button onClick={() => inserirNumero(9)}>9</button>
        <button onClick={() => inserirOperacao('+')}&plus;</button>
        
        <button onClick={() => inserirNumero(4)}>4</button>
        <button onClick={() => inserirNumero(5)}>5</button>
        <button onClick={() => inserirNumero(6)}>6</button>
        <button onClick={calcular} rowSpan="2">=</button>
        
        <button onClick={() => inserirNumero(1)}>1</button>
        <button onClick={() => inserirNumero(2)}>2</button>
        <button onClick={() => inserirNumero(3)}>3</button>
        
        <button onClick={() => inserirNumero(0)} style={{ gridColumn: 'span 2' }}>0</button>
        <button onClick={() => inserirNumero('.')}>.</button>
      </div>
    </div>
  );
}
