import { useNavigate } from 'react-router-dom';

function LoginForm() {
  const navigate = useNavigate();

  const handleLogin = async (credentials) => {
    try {
      await authAPI.login(credentials);
      // Redirecionar após login
      navigate('/dashboard');
      
      // Ou substituir a página atual (não volta com botão voltar)
      navigate('/dashboard', { replace: true });
      
      // Voltar uma página
      navigate(-1);
      
      // Avançar uma página
      navigate(1);
      
    } catch (error) {
      console.error('Login falhou:', error);
    }
  };

  return (
    <form onSubmit={handleLogin}>
      {/* campos do formulário */}
      <button type="submit">Entrar</button>
    </form>
  );
}

📋 useParams Hook


import { useParams } from 'react-router-dom';

function UserProfile() {
  const { id } = useParams();
  const [user, setUser] = useState(null);

  useEffect(() => {
    // Buscar usuário pelo ID
    fetchUser(id).then(setUser);
  }, [id]);

  if (!user) return <div>Carregando...</div>;

  return (
    <div>
      <h1>Perfil do Usuário</h1>
      <h2>{user.name}</h2>
      <p>ID: {id}</p>
      <p>Email: {user.email}</p>
    </div>
  );
}

function ProductDetail() {
  const { category, id } = useParams();
  
  return (
    <div>
      <h1>Produto {id}</h1>
      <p>Categoria: {category}</p>
    </div>
  );
}

🔍 Query Parameters


import { useSearchParams } from 'react-router-dom';

function ProductList() {
  const [searchParams, setSearchParams] = useSearchParams();
  
  // Ler parâmetros: /products?category=tech&sort=price&page=2
  const category = searchParams.get('category'); // 'tech'
  const sort = searchParams.get('sort'); // 'price'
  const page = parseInt(searchParams.get('page')) || 1; // 2
  
  const handleFilterChange = (newCategory) => {
    setSearchParams({
      category: newCategory,
      sort,
      page: 1 // Reset para primeira página
    });
  };
  
  const handleSortChange = (newSort) => {
    setSearchParams(prev => ({
      ...Object.fromEntries(prev),
      sort: newSort
    }));
  };

  return (
    <div>
      <h1>Produtos {category && `- ${category}`}</h1>
      
      <select onChange={(e) => handleFilterChange(e.target.value)}>
        <option value="">Todas as categorias</option>
        <option value="tech">Tecnologia</option>
        <option value="fashion">Moda</option>
      </select>
      
      <select onChange={(e) => handleSortChange(e.target.value)}>
        <option value="name">Nome</option>
        <option value="price">Preço</option>
      </select>
      
      <p>Página: {page}</p>
    </div>
  );
}

🔐 PrivateRoute Component


import { Navigate, useLocation } from 'react-router-dom';
import { useAuth } from '../contexts/AuthContext';

function PrivateRoute({ children }) {
  const { user, loading } = useAuth();
  const location = useLocation();

  if (loading) {
    return <div>Carregando...</div>;
  }

  if (!user) {
    // Salvar a localização que tentou acessar
    return <Navigate to="/login" state={{ from: location }} replace />;
  }

  return children;
}

// Uso nas rotas
function App() {
  return (
    <BrowserRouter>
      <Routes>
        <Route path="/login" element={<Login />} />
        <Route path="/register" element={<Register />} />
        
        <Route path="/dashboard" element={
          <PrivateRoute>
            <Dashboard />
          </PrivateRoute>
        } />
        
        <Route path="/profile" element={
          <PrivateRoute>
            <Profile />
          </PrivateRoute>
        } />
      </Routes>
    </BrowserRouter>
  );
}

👑 Role-Based Routes


function RoleRoute({ children, allowedRoles }) {
  const { user } = useAuth();
  const location = useLocation();

  if (!user) {
    return <Navigate to="/login" state={{ from: location }} />;
  }

  if (!allowedRoles.includes(user.role)) {
    return <Navigate to="/unauthorized" />;
  }

  return children;
}

// Uso
<Route path="/admin" element={
  <RoleRoute allowedRoles={['admin', 'super-admin']}>
    <AdminPanel />
  </RoleRoute>
} />

<Route path="/moderator" element={
  <RoleRoute allowedRoles={['admin', 'moderator']}>
    <ModeratorPanel />
  </RoleRoute>
} />

🔄 Redirecionamento após Login


import { useLocation, useNavigate } from 'react-router-dom';

function Login() {
  const navigate = useNavigate();
  const location = useLocation();
  
  // De onde o usuário veio antes do login
  const from = location.state?.from?.pathname || '/dashboard';

  const handleLogin = async (credentials) => {
    try {
      await authAPI.login(credentials);
      // Voltar para onde tentou acessar originalmente
      navigate(from, { replace: true });
    } catch (error) {
      console.error('Login failed:', error);
    }
  };

  return (
    <form onSubmit={handleLogin}>
      <h2>Login</h2>
      <p>Para acessar {from}, faça login primeiro.</p>
      {/* campos do form */}
    </form>
  );
}

import { lazy, Suspense } from 'react';

// Lazy loading dos componentes
const Home = lazy(() => import('./pages/Home'));
const About = lazy(() => import('./pages/About'));
const Dashboard = lazy(() => import('./pages/Dashboard'));
const UserProfile = lazy(() => import('./pages/UserProfile'));

// Loading component
function PageLoader() {
  return (
    <div className="page-loader">
      <div className="spinner"></div>
      <p>Carregando página...</p>
    </div>
  );
}

function App() {
  return (
    <BrowserRouter>
      <Suspense fallback={<PageLoader />}>
        <Routes>
          <Route path="/" element={<Home />} />
          <Route path="/about" element={<About />} />
          <Route path="/dashboard" element={<Dashboard />} />
          <Route path="/user/:id" element={<UserProfile />} />
        </Routes>
      </Suspense>
    </BrowserRouter>
  );
}

function App() {

    </BrowserRouter>
  );
}

📖 Página de Post


import { useParams, useNavigate } from 'react-router-dom';

function Post() {
  const { slug } = useParams();
  const navigate = useNavigate();
  const [post, setPost] = useState(null);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    const fetchPost = async () => {
      try {
        const postData = await blogAPI.getPostBySlug(slug);
        if (!postData) {
          navigate('/404', { replace: true });
          return;
        }
        setPost(postData);
      } catch (error) {
        console.error('Erro ao carregar post:', error);
      } finally {
        setLoading(false);
      }
    };

    fetchPost();
  }, [slug, navigate]);

  if (loading) return <PostSkeleton />;
  if (!post) return null;

  return (
    <article className="post">
      <header>
        <h1>{post.title}</h1>
        <div className="post-meta">
          <Link to={`/blog/author/${post.author.slug}`}>
            {post.author.name}
          </Link>
          <span>{formatDate(post.publishedAt)}</span>
          <Link to={`/blog/category/${post.category.slug}`}>
            {post.category.name}
          </Link>
        </div>
      </header>
      
      <div className="post-content">
        {post.content}
      </div>
      
      <footer>
        <div className="post-tags">
          {post.tags.map(tag => (
            <Link key={tag} to={`/blog?tag=${tag}`}>
              #{tag}
            </Link>
          ))}
        </div>
      </footer>
    </article>
  );
}

🧩 Breadcrumbs Component


import { Link, useLocation } from 'react-router-dom';

function Breadcrumbs() {
  const location = useLocation();
  const pathnames = location.pathname.split('/').filter(x => x);

  return (
    <nav className="breadcrumbs">
      <Link to="/">Home</Link>
      {pathnames.map((value, index) => {
        const to = `/${pathnames.slice(0, index + 1).join('/')}`;
        const isLast = index === pathnames.length - 1;

        return isLast ? (
          <span key={to}> / {formatBreadcrumb(value)}</span>
        ) : (
          <span key={to}>
            {' / '}
            <Link to={to}>{formatBreadcrumb(value)}</Link>
          </span>
        );
      })}
    </nav>
  );
}

function formatBreadcrumb(value) {
  return value.charAt(0).toUpperCase() + value.slice(1);
}
