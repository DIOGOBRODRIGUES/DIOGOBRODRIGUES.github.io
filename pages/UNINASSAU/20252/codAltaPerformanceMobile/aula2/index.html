🚀 Por que React Native?

  • 📱 Nativo Real: Não é WebView, são componentes nativos
  • ⚡ Performance: 60fps com otimizações adequadas
  • 🔄 Code Sharing: 90% do código entre iOS e Android
  • 👨‍💻 Produtividade: Hot Reload e Fast Refresh
  • 🌍 Comunidade: Usado por Facebook, Instagram, Uber, Tesla

import { TextInput, View, Text, StyleSheet } from 'react-native';
import { useState } from 'react';

function CustomInput({ label, placeholder, value, onChangeText, ...props }) {
  const [isFocused, setIsFocused] = useState(false);

  return (
    <View style={styles.container}>
      {label && <Text style={styles.label}>{label}</Text>}
      <TextInput
        style={[styles.input, isFocused && styles.inputFocused]}
        placeholder={placeholder}
        value={value}
        onChangeText={onChangeText}
        onFocus={() => setIsFocused(true)}
        onBlur={() => setIsFocused(false)}
        placeholderTextColor="#999"
        {...props}
      />
    </View>
  );
}

  container: {
    marginVertical: 8,
  },

📐 Dimensions e Responsive Design


import { Dimensions, PixelRatio } from 'react-native';

// Obter dimensões da tela
const { width, height } = Dimensions.get('window');
const screenData = Dimensions.get('screen');

// Função para design responsivo
const responsiveWidth = (percentage) => {
  return (percentage * width) / 100;
};

const responsiveHeight = (percentage) => {
  return (percentage * height) / 100;
};

// Pixel perfeito
const normalize = (size) => {
  const scale = width / 375; // iPhone X como base
  const newSize = size * scale;
  
  if (PixelRatio.get() >= 2 && newSize >= 12) {
    return Math.round(PixelRatio.roundToNearestPixel(newSize));
  }
  return Math.round(newSize);
};

const styles = StyleSheet.create({
  responsiveContainer: {
    width: responsiveWidth(90), // 90% da largura
    height: responsiveHeight(30), // 30% da altura
  },
  responsiveText: {
    fontSize: normalize(16), // Tamanho adaptativo
  }
});

🎯 Otimizações de Estado


import { useState, useEffect, useCallback, useMemo } from 'react';

function OptimizedComponent({ data }) {
  const [searchTerm, setSearchTerm] = useState('');
  const [isLoading, setIsLoading] = useState(false);

  // Memoiza dados filtrados
  const filteredData = useMemo(() => {
    if (!searchTerm) return data;
    return data.filter(item => 
      item.name.toLowerCase().includes(searchTerm.toLowerCase())
    );
  }, [data, searchTerm]);

  // Callback memoizado para performance
  const handleSearch = useCallback((text) => {
    setSearchTerm(text);
  }, []);

  // Debounce para busca
  useEffect(() => {
    const timer = setTimeout(() => {
      if (searchTerm) {
        setIsLoading(true);
        // Simular busca na API
        setTimeout(() => setIsLoading(false), 500);
      }
    }, 300);

    return () => clearTimeout(timer);
  }, [searchTerm]);

  return (
    <View>
      <TextInput
        placeholder="Buscar..."
        value={searchTerm}
        onChangeText={handleSearch}
      />
      {isLoading ? (
        <ActivityIndicator size="large" />
      ) : (
        <FlatList
          data={filteredData}
          renderItem={renderOptimizedItem}
          keyExtractor={keyExtractor}
        />
      )}
    </View>
  );
}

🔄 Navegação entre Telas


function HomeScreen({ navigation }) {
  const navigateToProfile = () => {
    navigation.navigate('Profile', {
      userId: 123,
      userName: 'João Silva'
    });
  };

  return (
    <View style={styles.container}>
      <Text style={styles.title}>Tela Inicial</Text>
      <TouchableOpacity 
        style={styles.button} 
        onPress={navigateToProfile}
      >
        <Text style={styles.buttonText}>Ver Perfil</Text>
      </TouchableOpacity>
    </View>
  );
}

function ProfileScreen({ route, navigation }) {
  const { userId, userName } = route.params;

  return (
    <View style={styles.container}>
      <Text>Perfil de {userName}</Text>
      <Text>ID: {userId}</Text>
      <TouchableOpacity onPress={() => navigation.goBack()}>
        <Text>Voltar</Text>
      </TouchableOpacity>
    </View>
  );
}
